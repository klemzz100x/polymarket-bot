def bayes_update(prior, likelihood_if_true, likelihood_overall):
    posterior = (likelihood_if_true * prior) / likelihood_overall
    return round(posterior, 3)

# Prior: 0.40 (base rate)

def detect_insiders(prices_df, window=12, k=3.0):
    prices_df['H']     = prices_df['price'].apply(entropy)
    prices_df['dH']    = prices_df['H'].diff()
    prices_df['sigma'] = prices_df['dH'].rolling(window).std()
