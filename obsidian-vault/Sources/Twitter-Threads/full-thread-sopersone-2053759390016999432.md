Why Machine Learning in Trading
Classical strategies have one great strength: they are understandable. You know exactly why the algorithm entered a position. But they have a fundamental limitation: the rules are invented by a human. And humans can miss patterns that are not obvious to the eye.
The ML approach inverts the logic: we do not tell the model what to look for. We hand it data and say: "Find whatever predicts price movement."
This opens up new possibilities, but also new risks. The main one is overfitting: the model memorizes noise instead of signal. We will cover how to fight that.
Three Tasks, Three Approaches
In trading, ML is used for three types of problems:
- **Regression**: predict a numerical value - tomorrow's price or return
- **Classification**: predict direction - will the price go up or down (0 or 1)
- **Anomaly detection and clustering**: find unusual market regimes
In this article we cover the first two. The third will come later.

def make_features(df, lags=5):
    """
    Build features from lagged log returns.

    Parameters:
        df   -- DataFrame with a 'log_return' column
        lags -- number of lags

    Returns:
        DataFrame with features and target variable

    data.dropna(inplace=True)

    return data

def ml_backtest(data, y_pred_cls, test_index):
    """
    Simple backtest of an ML strategy.

    Parameters:
        data       -- original DataFrame with log_return
        y_pred_cls -- predicted classes (0 or 1) on the test period
        test_index -- index of the test set

    Returns:
        result -- DataFrame with results
    """
    result = data.loc[test_index].copy()

    # Position: 1 if the model predicts up, 0 if down
    result["position"] = y_pred_cls
    # Shift by 1 day (we trade the next day)
    result["position"] = result["position"].shift(1)

    result.dropna(subset=["position"], inplace=True)

    # Strategy return
    result["strategy_return"]   = result["log_return"] * result["position"]

    # Cumulative returns
    result["cumret_strategy"]   = result["strategy_return"].cumsum().apply(np.exp)
    result["cumret_buyandhold"] = result["log_return"].cumsum().apply(np.exp)

    return result

def make_rich_features(df, lags=5):
    """
    Extended feature engineering with technical indicators.
    """
    data = df[["close", "volume", "log_return"]].copy()

    # Return lags
    for lag in range(1, lags + 1):
        data[f"lag_{lag}"] = data["log_return"].shift(lag)

    # Moving average ratio
    data["sma_20"] = data["close"].rolling(20).mean()
    data["sma_50"] = data["close"].rolling(50).mean()
    data["sma_ratio"] = data["sma_20"] / data["sma_50"] - 1  # >0: bullish signal

    # Momentum
    data["momentum_10"] = np.log(data["close"] / data["close"].shift(10))
    data["momentum_20"] = np.log(data["close"] / data["close"].shift(20))

    # Realized volatility over the last 10 days
    data["volatility_10"] = data["log_return"].rolling(10).std()

    # Z-score of price relative to 20-day mean
    rolling_mean = data["close"].rolling(20).mean()
    rolling_std  = data["close"].rolling(20).std()
    data["zscore_20"] = (data["close"] - rolling_mean) / rolling_std

    # Relative volume
    data["volume_ratio"] = data["volume"] / data["volume"].rolling(20).mean()

    # Target
    data["target_direction"] = (data["log_return"].shift(-1) > 0).astype(int)

    data.dropna(inplace=True)

    return data

A Word on Overfitting
The more expressive the model, the higher the risk of memorizing the training data. A random forest with 500 trees will show 99% accuracy on train and 51% on test. That is a textbook example of overfitting.
Signs of overfitting:
- A large gap between train and test accuracy
- The model performs below baseline on test
- Adding more features improves train but not test
How to fight it: fewer parameters, regularization, cross-validation, walk-forward testing.

 Walk-Forward Validation for ML
In Article 6 we covered walk-forward analysis for classical strategies. For ML it is even more important. A simple two-chunk train/test split is an overly optimistic estimate of real performance.
Walk-forward for ML: a rolling training window, with predictions made one step at a time.
python
def walk_forward_ml(data, feature_cols, target_col, train_size=252, step=21):
    """
    Walk-forward validation of an ML classifier.

    Parameters:
        data         -- DataFrame with features and target
        feature_cols -- list of feature column names
        target_col   -- name of the target column
        train_size   -- training window size (days)
        step         -- rolling step (days)

    Returns:
        DataFrame with predictions and actual values
    """
    X    = data[feature_cols].values
    y    = data[target_col].values
    idx  = data.index

    all_preds  = []
    all_actual = []
    all_dates  = []

    for start in range(0, len(data) - train_size - step, step):
        end_train = start + train_size
        end_test  = end_train + step

        X_tr, y_tr = X[start:end_train], y[start:end_train]
        X_te, y_te = X[end_train:end_test], y[end_train:end_test]

        sc   = StandardScaler()
        X_tr = sc.fit_transform(X_tr)
        X_te = sc.transform(X_te)

        clf  = LogisticRegression(random_state=42, max_iter=500)
        clf.fit(X_tr, y_tr)
        preds = clf.predict(X_te)

        all_preds.extend(preds)
        all_actual.extend(y_te)
        all_dates.extend(idx[end_train:end_test])

    wf_result = pd.DataFrame({
        "date":      all_dates,
        "actual":    all_actual,
        "predicted": all_preds,
    }).set_index("date")

    wf_acc = accuracy_score(wf_result["actual"], wf_result["predicted"])
    print(f"Walk-forward accuracy: {wf_acc:.1%}")

    return wf_result

wf = walk_forward_ml(data_rich, feature_cols_rich, "target_direction",
                     train_size=252, step=21)

Walk-forward accuracy is typically lower than accuracy on a fixed test set. That is the honest estimate of what to expect in live trading.

ml_strategy.py -- ML strategy with logistic regression on BTC
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression, LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score
from data_loader import fetch_btc_data, add_returns


def make_features(df, lags=5):
    data = df[["close", "volume", "log_return"]].copy()
    for lag in range(1, lags + 1):
        data[f"lag_{lag}"] = data["log_return"].shift(lag)
    data["sma_ratio"]     = df["close"].rolling(20).mean() / df["close"].rolling(50).mean() - 1
    data["momentum_10"]   = np.log(df["close"] / df["close"].shift(10))
    data["momentum_20"]   = np.log(df["close"] / df["close"].shift(20))
    data["volatility_10"] = data["log_return"].rolling(10).std()
    rm = df["close"].rolling(20).mean()
    rs = df["close"].rolling(20).std()
    data["zscore_20"]     = (df["close"] - rm) / rs
    data["volume_ratio"]  = df["volume"] / df["volume"].rolling(20).mean()
    data["target"]        = (data["log_return"].shift(-1) > 0).astype(int)
    data.dropna(inplace=True)
    return data


def run_ml_backtest(data, feature_cols, train_ratio=0.7):
    split_idx = int(len(data) * train_ratio)
    X_train = data[feature_cols].iloc[:split_idx]
    X_test  = data[feature_cols].iloc[split_idx:]
    y_train = data["target"].iloc[:split_idx]
    y_test  = data["target"].iloc[split_idx:]

    sc     = StandardScaler()
    Xtr_sc = sc.fit_transform(X_train)
    Xte_sc = sc.transform(X_test)

    clf    = LogisticRegression(random_state=42, max_iter=1000)
    clf.fit(Xtr_sc, y_train)
    preds  = clf.predict(Xte_sc)

    acc    = accuracy_score(y_test, preds)

    result = data.iloc[split_idx:].copy()
    result["position"]        = pd.Series(preds, index=X_test.index).shift(1)
    result.dropna(subset=["position"], inplace=True)
    result["strategy_return"] = result["log_return"] * result["position"]
    result["cumret_strategy"] = result["strategy_return"].cumsum().apply(np.exp)
    result["cumret_bh"]       = result["log_return"].cumsum().apply(np.exp)

    total_ret  = result["cumret_strategy"].iloc[-1] - 1
    annual_vol = result["strategy_return"].std() * np.sqrt(252)
    annual_ret = (1 + total_ret) ** (252 / len(result)) - 1
    sharpe     = annual_ret / annual_vol if annual_vol > 0 else 0
    rm         = result["cumret_strategy"].cummax()
    max_dd     = ((result["cumret_strategy"] - rm) / rm).min()

    print(f"Accuracy:        {acc:.1%}")
    print(f"Baseline:        {y_test.mean():.1%}")
    print(f"Strategy return: {total_ret:+.1%}")
    print(f"Buy & Hold:      {result['cumret_bh'].iloc[-1] - 1:+.1%}")
    print(f"Sharpe ratio:    {sharpe:.2f}")
    print(f"Max drawdown:    {max_dd:.1%}")

    return result, acc
