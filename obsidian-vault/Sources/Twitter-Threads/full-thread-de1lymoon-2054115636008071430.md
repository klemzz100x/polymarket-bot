def shannon_entropy(p):
    """
    Calculate Shannon entropy in bits for a binary outcome.

    p: probability of YES (0 to 1)
    Returns: entropy in bits (0 to 1)
    """
    p = np.clip(p, 0.001, 0.999)

    return -(p * np.log2(p) + q * np.log2(q))

def kl_divergence(p_mine, p_market):
    """
    Measure edge in bits.

    p_mine: your probability estimate
    p_market: current market price
    Returns: KL divergence in bits

    Interpretation:
    < 0.05 bits: No edge. Skip.
    0.05-0.10 bits: Weak edge. Fees may eat it.
    0.10-0.20 bits: Real edge. Trade.
    > 0.30 bits: Check your math. Something is wrong.
    """
    p_mine = np.clip(p_mine, 0.01, 0.99)
    p_market = np.clip(p_market, 0.01, 0.99)

    q_mine = 1 - p_mine
    q_market = 1 - p_market

    return (p_mine * np.log2(p_mine / p_market) +
            q_mine * np.log2(q_mine / q_market))

def scan_markets(my_estimates, market_prices, market_names):
    """
    Scan multiple markets, rank by KL-divergence edge.
    """
    results = []
    for name, p_mine, p_market in zip(market_names, my_estimates, market_prices):
        kl = kl_divergence(p_mine, p_market)
        edge_cents = (p_mine - p_market) * 100

        if kl < 0.05:
            signal = "SKIP — no edge"
        elif kl < 0.10:
            signal = "WEAK — fees may kill it"
        elif kl < 0.20:
            signal = "TRADE — real edge"
        else:
            signal = "RECHECK — suspiciously high"

        results.append({
            "market": name,
            "my_estimate": p_mine,
            "market_price": p_market,
            "edge_cents": edge_cents,
            "kl_bits": kl,
            "signal": signal
        })

    return sorted(results, key=lambda x: x["kl_bits"], reverse=True)

import numpy as np
from scipy import stats
from scipy.stats import kurtosis, norm

def fat_tail_analysis(historical_returns, confidence=0.95):
    """
    Analyze whether a market's price history has fat tails
    and by how much Gaussian models underestimate extreme events.

    historical_returns: list of price changes per period
    confidence: VaR confidence level
    """
    returns = np.array(historical_returns)

    # Basic statistics
    mu = np.mean(returns)
    sigma = np.std(returns)

    # Kurtosis (normal distribution = 3, fat tails > 3)
    kurt = kurtosis(returns, fisher=False)  # Pearson's kurtosis
    excess_kurt = kurt - 3

    # Gaussian VaR vs actual VaR
    gaussian_var = norm.ppf(1 - confidence, mu, sigma)
    empirical_var = np.percentile(returns, (1 - confidence) * 100)

    # How much does Gaussian underestimate tail risk?
    tail_ratio = abs(empirical_var / gaussian_var) if gaussian_var != 0 else 1

    return {
        "kurtosis": kurt,
        "excess_kurtosis": excess_kurt,
        "fat_tails_present": kurt > 3,
        "gaussian_var_95": gaussian_var,
        "empirical_var_95": empirical_var,
        "gaussian_underestimates_by": f"{(tail_ratio - 1) * 100:.0f}%"
    }

def barbell_portfolio(bankroll, safe_markets, tail_markets):
    """
    Build a Taleb barbell on Polymarket.

    safe_markets: list of (price_cents, true_probability) for high-prob markets
    tail_markets: list of (price_cents, true_probability) for tail-risk markets
    """
    safe_allocation = bankroll * 0.85
    tail_allocation = bankroll * 0.15

    # Equal-weight within each bucket
    safe_per_market = safe_allocation / len(safe_markets)
    tail_per_market = tail_allocation / len(tail_markets)

    portfolio = {"safe": [], "tail": []}

    for price, true_prob in safe_markets:
        ev = (true_prob * (100 - price) - (1 - true_prob) * price) / price
        portfolio["safe"].append({
            "price": price,
            "true_prob": true_prob,
            "allocation": safe_per_market,
            "expected_ev": ev * safe_per_market
        })

    for price, true_prob in tail_markets:
        payout_multiple = (100 - price) / price  # e.g., 3¢ contract → 32x payout
        ev = true_prob * payout_multiple - (1 - true_prob)
        portfolio["tail"].append({
            "price": price,
            "true_prob": true_prob,
            "allocation": tail_per_market,
            "payout_multiple": payout_multiple,
            "expected_ev": ev * tail_per_market
        })

    return portfolio
