import numpy as np

def shannon_entropy(p):
    """Returns entropy in bits for a binary outcome."""
    if p <= 0 or p >= 1:
        return 0.0
    q = 1 - p
    return -(p * np.log2(p) + q * np.log2(q))

# Fair market (max uncertainty)
print(shannon_entropy(0.50))  # 1.000 bits
print(shannon_entropy(0.35))  # 0.934 bits
print(shannon_entropy(0.95))  # 0.286 bits
Shannon entropy H(p) for any binary Polymarket outcome. The curve peaks at exactly 1 bit when p = 0.5 - that's the most uncertainty a binary event can hold. import numpy as np

def kl_divergence_binary(p_mine, p_market):
    """Returns KL divergence in bits for a binary market.
       p_mine   — your estimate (0..1)
       p_market — market price (0..1)
    """
    # Clip to avoid log(0)
    p_mine   = np.clip(p_mine,   0.01, 0.99)
    p_market = np.clip(p_market, 0.01, 0.99)

    q, qm = 1 - p_mine, 1 - p_market
    return (p_mine * np.log2(p_mine / p_market)
            + q * np.log2(q / qm))

# Your estimate 0.55, market price 0.35
edge = kl_divergence_binary(0.55, 0.35)
print(f"Edge: {edge:.3f} bits")  # 0.120 bits

Common mistakes
KL is asymmetric: D_KL(P∥Q) ≠ D_KL(Q∥P). First arg is always your estimate.
Breaks on p=0 or p=1. Always clip to [0.01, 0.99].
High D_KL ≠ guaranteed profit. Fees, liquidity, slippage still matter.

import numpy as np
from scipy.optimize import minimize

def max_entropy_fusion(signals, confidences):
    """
    signals     — list of probability estimates [p1, p2, ..., pN]
    confidences — list of weights (higher = more trust)
    Returns a single fused probability via max-entropy weighting.
    """
    signals     = np.array(signals)
    confidences = np.array(confidences)

    # Constraint: weighted mean matches signal consensus
    target = np.sum(signals * confidences) / np.sum(confidences)

    # Max-entropy under mean constraint on {0,1} = Bernoulli(target)
    return np.clip(target, 0.01, 0.99)

signals     = [0.40, 0.60, 0.50]
confidences = [1.0,  2.0,  1.5]  # news most trusted

fused = max_entropy_fusion(signals, confidences)
print(f"Fused probability: {fused:.3f}")  # ~0.522

import numpy as np
import pandas as pd

def entropy(p):
    p = np.clip(p, 0.01, 0.99)
    return -(p * np.log2(p) + (1-p) * np.log2(1-p))

def detect_insiders(prices_df, window=12, k=3.0):
    """
    prices_df : DataFrame with columns ['timestamp', 'price']
                (5-minute intervals recommended)
    window    : rolling window for baseline stats
    k         : sigma threshold
    """
    prices_df['H']    = prices_df['price'].apply(entropy)
    prices_df['dH']   = prices_df['H'].diff()
    prices_df['sigma'] = prices_df['dH'].rolling(window).std()
    prices_df['alert'] = np.abs(prices_df['dH']) > k * prices_df['sigma']

    return prices_df[prices_df['alert']]
